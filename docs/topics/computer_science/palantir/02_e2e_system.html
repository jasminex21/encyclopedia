# pipeline.py — runs every night at 04:00 London time
from datetime import datetime, timedelta
from palantir_models import Trip, Zone, Recommendation
from palantir_foundry import ontology_manager, datasets

def main():
    # 1. Load the last 24 hours of cleaned trips
    raw_trips = datasets.get("cleaned_taxi_trips").as_pandas()
    
    # 2. Do the heavy math — calculate £/hour for every zone for the next 4-hour windows
    zone_scores = calculate_zone_scores(raw_trips)          # ← your 20-line scoring function
    
    # 3. Write new/updated Zone objects back to the ontology
    zones_to_write = []
    for zone_id, score in zone_scores.items():
        zone = Zone.find(zone_id) or Zone(zone_id=zone_id, name=zone_id)
        zone.current_eph_score = score
        zone.last_calculated   = datetime.utcnow()
        zones_to_write.append(zone)
    
    # 4. One line that does all the magic
    ontology_manager.write(zones_to_write)
    
    print(f"Updated {len(zones_to_write)} zones at {datetime.utcnow()}")

if __name__ == "__main__":
    main()

# airflow/dags/daily_zone_recommendations.py
from airflow import DAG
from airflow.operators.python import PythonOperator
from datetime import datetime, timedelta
import pandas as pd
import sqlalchemy

default_args = {"retries": 3, "retry_delay": timedelta(minutes=5)}

with DAG(
    dag_id="daily_zone_recommendations",
    schedule="0 4 * * *",          # 04:00 UTC every day
    start_date=datetime(2025, 1, 1),
    catchup=False,
    default_args=default_args,
    tags=["taxi", "production"],
) as dag:

    # Task 1 – heavy math (exactly like Code Repository)
    def calculate_raw_scores():
        engine = sqlalchemy.create_engine("postgresql://...")
        trips = pd.read_sql("SELECT * FROM fct_trip WHERE started_at >= current_date - 7", engine)
        scores = trips.groupby("pickup_zone_id").apply(compute_eph_with_bonuses)
        scores.to_sql("temp_raw_zone_scores", engine, if_exists="replace")

    raw_scores_task = PythonOperator(task_id="calculate_raw_scores", python_callable=calculate_raw_scores)

    # Task 2 – business rules & human gate (poor-man’s AIP Logic)
    def apply_business_rules_and_write_recommendations():
        # same engine...
        raw = pd.read_sql("SELECT * FROM temp_raw_zone_scores", engine)
        
        # ← these three lines are the “non-engineer” part in real life
        AIRPORT_BONUS = 1.4          # ← fleet manager changes this in a tiny admin table instead
        MAX_DRIVERS_PER_ZONE = 3
        
        final = raw.copy()
        final["final_score"] = final["raw_score"]
        final.loc[final.zone_name.str.contains("Heathrow"), "final_score"] *= AIRPORT_BONUS
        
        # simple human-in-the-loop via a “pending” flag + Slack message
        high_risk = final[final["final_score"] > 60]
        if not high_risk.empty:
            send_slack_message("@captain-johnson please approve these crazy zones")
            final.loc[final["final_score"] > 60, "needs_approval"] = True

        # write final recommendations
        final.head(50).to_sql("fct_recommendation", engine, if_exists="append")

    final_task = PythonOperator(task_id="apply_rules_and_write", python_callable=apply_business_rules_and_write_recommendations)

    raw_scores_task >> final_task
